# Automation Readiness And Needs You Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Move Automation readiness out of Opportunities/Analysis, make Settings Automation the actionable setup home, and feed Home `Needs you` with a deterministic P0/P1/P2 global action queue.

**Architecture:** Keep the existing Autopilot readiness API as the canonical source, add persistent recovery-plan acknowledgement to `seo_policies`, and introduce one shared frontend readiness-action mapper so Home, Settings, and Analysis use the same gate-to-CTA contract. Settings owns policy/readiness controls; Home renders unresolved user actions; Analysis keeps opportunity review as the primary surface and only shows a lightweight Settings bridge when automation setup is incomplete.

**Tech Stack:** Go HTTP handlers, pgx/sqlc queries, PostgreSQL migrations, Next.js App Router, React client components, TypeScript, Tailwind CSS, Node contract tests, Go tests.

export function toneForHumanPriority(priority: HumanActionPriority): HumanActionTone {
  if (priority === "P0") return "red";
  if (priority === "P1") return "amber";
  return "blue";
}

export function buildReadinessHumanActions(input: {
  projectId: string;
  readiness: AutopilotReadiness | null;
  canAccessSettings: boolean;
}): ReadinessHumanAction[] {
  const { projectId, readiness, canAccessSettings } = input;
  if (!readiness || !canAccessSettings) return [];

  return readiness.gates
    .filter((gate): gate is AutopilotReadinessGate => Boolean(gate?.blocking))
    .map((gate) => {
      const action = READINESS_GATE_ACTIONS[gate.key];
      if (!action) return null;
      const priority = readiness.autopilot_level >= 2 ? action.priorityWhenLevel2 : action.priorityBeforeLevel2;
      const detail =
        gate.key === "monthly_budget_configured"
          ? `${gate.reason} Set SEOPolicy.monthly_budget_limit in Automation policy.`
          : gate.reason;
      return {
        id: `readiness-${gate.key}`,
        dedupeKey: `readiness:${gate.key}`,
        title: action.title,
        detail,
        href: action.href(projectId),
        cta: action.cta,
        category: priority === "P0" ? "Blocking now" : action.category,
        priority,
        rank: action.rank,
        tone: toneForHumanPriority(priority),
      };
    })
    .filter((item): item is ReadinessHumanAction => Boolean(item));
}

export function dedupeHumanActions<T extends { dedupeKey?: string; id: string }>(items: T[]): T[] {
  const seen = new Set<string>();
  const deduped: T[] = [];
  for (const item of items) {
    const key = item.dedupeKey ?? item.id;
    if (seen.has(key)) continue;
    seen.add(key);
    deduped.push(item);
  }
  return deduped;
}
